				function getNum(str){
					var arr = [];
					var res = ""
					for(var i=0;i<str.length;i++){
						if(str.charAt(i)>="0" && str.charAt(i)<="9"){
							res +=str.charAt(i);
						}else{
							//排除掉空字符
							if(res){
								arr.push(res);
								//每次push完一个整体的数字字符就清空字符res
								res = "";
							}	
						}
					}
					//防止字符串str最后一个字符为数字没有放进数组
					if(res){
						arr.push(res);
					}
					return arr;
				};
				console.log(getNum(str));
